	public void tick(ArrayList<Entity> ent){
		x += dx;
		y += dy;
		
		dx = 0;
		dy = 0;
		
		stopFalling(ent);
		
		if (KeyBoard.down && onGround(ent) && !falling) {
			dy = Constant.GRAVITY;
		}
		if (KeyBoard.up && !falling){
			dy = JUMP_SPEED;			
		}
		if (KeyBoard.left ){
			dx = -VEL;
		}
		if (KeyBoard.right){
			dx = VEL;
		}			
		if (!onGround(ent)){
			falling = true;
			dy = Constant.GRAVITY;
		}
		

	}
	
	private boolean onGround(ArrayList <Entity> ent){
		for (int i = 0; i < ent.size(); i++){
			Entity e = ent.get(i);
			if (e.getID() == ID.Building) {
				return getBotBounds().intersects(e.getBotBounds());
			}
		}
		return false;
	}
	
	private void stopFalling(ArrayList<Entity> ent){
		if (onGround(ent) && falling){
			falling = false;
		}
	}
